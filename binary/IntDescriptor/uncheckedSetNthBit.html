T uncheckedSetNthBit(int v, int n, [bool value = true]) {
  if (signed) {
    v = v.toUnsigned(width);
  }
  if (value) {
    v |= 1 << n;
  } else {
    v &= ~(1 << n);
  }
  if (signed) {
    v = v.toSigned(width);
  }
  return _uncheckedCast(v);
}
